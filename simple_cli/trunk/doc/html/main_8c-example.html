#include <stdio.h>
#include <string.h>
#include "simple_cli.h"
// User function 1.
void fun1(char *response, size_t responseSize, char **par)
{
char *par1 = *(par + 0);
char *par2 = *(par + 1);
_snprintf(response, responseSize - 1, "fun1 : <%s><%s>\r\n", par1, par2);
}
// User function 2.
void fun2(char *response, size_t responseSize, char **par)
{
char *par1 = *(par + 0);
char *par2 = *(par + 1);
char *par3 = *(par + 2);
char *par4 = *(par + 3);
_snprintf(response, responseSize - 1, "fun2 : <%s><%s><%s><%s>\r\n", par1, par2, par3, par4);
}
int main(void)
{
cli_t cli;
CliInit(&cli);
CliRegisterCommand(&cli, "fun1", fun1, 2);
CliRegisterCommand(&cli, "fun2", fun2, 4);
char cmd[] = "fun1 one two";
char *response = NULL;
response = CliParseCommand(&cli, cmd);
if (response != NULL)
{
printf("%s\r\n", response);
}
else
{
printf("no cmd found\r\n");
}
getchar();
return 0;
}
